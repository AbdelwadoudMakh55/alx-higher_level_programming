``1-my_list`` module
===============================
using ``print_sorted``
===============================

``print_sorted()`` print a list in order.

::

    >>> MyList = __import__('1-my_list').MyList
    >>> my_list = MyList()
    >>> my_list.append(0)
    >>> my_list.append(14)
    >>> my_list.append(-9)
    >>> my_list.append(3)
    >>> my_list.print_sorted()
    [-9, 0, 3, 14]

::

    >>> my_list = MyList()
    >>> my_list.append(0)
    >>> my_list.append(0)
    >>> my_list.append(0)
    >>> my_list.append(0)
    >>> my_list.print_sorted()
    [0, 0, 0, 0]

::

    >>> my_list = MyList()
    >>> my_list.append(0)
    >>> my_list.append(1)
    >>> my_list.append(2)
    >>> my_list.append(3)
    >>> my_list.append(4)
    >>> my_list.print_sorted()
    [0, 1, 2, 3, 4]

::

    >>> my_list = MyList()
    >>> my_list.print_sorted()
    []

::
   >>> my_list = MyList()
   >>> my_list.append(10)
   >>> my_list.print_sorted()
   [10]

::
   >>> my_list = MyList()
   >>> my_list.append(10)
   >>> my_list.append(9)
   >>> my_list.append(8)
   >>> my_list.append(5)
   >>> my_list.print_sorted()
   [5, 8, 9, 10]
